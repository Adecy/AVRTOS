	CONFIG_KERNEL_STATS_FIFO=1
	CONFIG_KERNEL_STATS_MSGQ=1
	CONFIG_KERNEL_STATS_TIMER=1
	CONFIG_KERNEL_STATS_EVENT=1
	CONFIG_KERNEL_STATS_WORKQ=1

)

target_link_avrtos(${PROJECT_NAME})

target_prepare_env(${PROJECT_NAME})
